module Base
{
    /* =========================================================
       MODELLE (je Stufe eigenes Rad)
       FBX in media/models_X/: holzwagen_t1 (Vollholzrad),
       holzwagen_t2 (Speichenrad). Textur gemeinsam: holzwagen_tex.png
       Modell ist in echten Metern gebaut (Wagen ~2,2 m). PZ liest die FBX-Einheiten
       NICHT als Meter, daher scale = 0.006 (sonst ~1000x zu gross).
       Feintuning: zu klein -> 0.002/0.005, zu gross -> 0.0008/0.0005.
       ========================================================= */
    /* ===== POSITION DES WAGENS IN DER HAND EINSTELLEN =====
       attachment Bip01_Prop1 = Lage des Modells IN DER HAND (eigene Kategorie,
       hat NICHTS mit der Map-Platzierung zu tun).
         rotate = a  b  c   (Grad). Empirisch (Blender Z-up -> PZ):
            a = kippen vor/zurueck   (Nase hoch/runter)
            b = drehen um die Hochachse / Yaw  <-- DAS ist die "Z-Drehung"
            c = kippen seitlich / Roll
         offset = X  Y  Z   (Meter)  verschieben.
       Aktuell: b = 90 = 90 Grad um die Hochachse, Uhrzeigersinn.
       Falsch herum -> -90.
       Ausfuehrliche Anleitung steht in HolzwagenMod/README_INSTALL.md. */
    /* ===== ZWEI LAGEN =====
       attachment Bip01_Prop1 = IN DER HAND. a (1. rotate-Wert) kippt den
         Handgriff hoch Richtung Haende (negativ = hoch; Vorzeichen tauschen,
         falls falsch herum). offset y hebt den ganzen Wagen.
       attachment world = ABGESTELLT auf dem Boden. a kippt den Wagen auf der
         Radachse, sodass der Handgriff nach unten auf den Boden sinkt. */
    model wagenT1
    {
        /* _hand = 10 Grad Griff-HOCH fest in die Geometrie gebacken
           (tools/holzwagen_world_tilt.py, HAND_UP_DEG). Deshalb hier a = 0. */
        mesh = holzwagen_t1_hand, texture = holzwagen_tex, scale = 0.007,
        attachment Bip01_Prop1 { offset = -1.4 0.1 0.0, rotate = 0.0 90.0 0.0, }
    }
    /* ABGESTELLT: Kippung (15.8 Grad um die Radachse) fest in die Geometrie
       gebacken -> Handgriff liegt auf dem Boden, Raeder bleiben am Boden.
       Erzeugt von tools/holzwagen_world_tilt.py aus dem funktionierenden FBX
       (gleicher Export-Weg, NUR die Vertices gedreht). Nur als WorldStaticModel! */
    model wagenT1world
    {
        mesh = holzwagen_t1_world, texture = holzwagen_tex, scale = 0.007,
    }
    /* T2 nutzt das Motion-Blur-Rad: die Speichen sind radial verwischt in die
       Textur holzwagen_t2_blur gebacken (eigener Atlas) und auf flache
       Rad-Scheiben gelegt -> sieht statisch aus wie in Drehung. Animierte
       Hand-Modelle gehen in PZ NICHT (ModelInstance teilt den AnimPlayer des
       Spielers). Erzeugt von tools/holzwagen_t2_blur.py. */
    model wagenT2
    {
        /* _hand = 10 Grad Griff-HOCH gebacken -> a = 0 (siehe wagenT1). */
        mesh = holzwagen_t2_blur_hand, texture = holzwagen_t2_blur, scale = 0.007,
        attachment Bip01_Prop1 { offset = -1.4 0.1 0.0, rotate = 0.0 90.0 0.0, }
    }
    /* ABGESTELLT: gebackene 15.8-Grad-Kippung (siehe wagenT1world). */
    model wagenT2world
    {
        mesh = holzwagen_t2_blur_world, texture = holzwagen_t2_blur, scale = 0.007,
    }

    /* =========================================================
       WAGEN ALS AUSRUESTBARES CONTAINER-ITEM (Schiebe-Prinzip)
       Engine-Felder wie bei SaucedCarts (funktionierender B42-Mod):
       - RunSpeedModifier = Tempo-Bremse (Engine)
       - StaticModel + ReplaceInPrimaryHand/SecondHand mit Schiebe-Maske
         (holdingtrolleyright/left) = Wagen sichtbar in den Haenden + Schiebe-Pose.
       Das Anschirren/Schieben macht Holzwagen_CartEquip.lua (Maske setzen).
       Volumen via Holzwagen_CapacityFix.lua zur Laufzeit (Script-Cap ist gedeckelt).
       Kein Engine-Fahrzeug.
       ========================================================= */
    item Holzwagen_T1
    {
        DisplayName      = Holzwagen (T1),
        ItemType         = base:container,   /* B42-Container-Vorlage: oeffenbar wie ein Rucksack */
        Weight           = 12.0,
        Capacity         = 50,
        WeightReduction  = 100,
        AcceptItemFunction = HolzwagenAccept,
        CloseSound       = CloseBag,
        OpenSound        = OpenBag,
        PutInSound       = PutItemInBag,
        WorldStaticModel = wagenT1world,  /* abgestellt: Handgriff am Boden */
        StaticModel      = wagenT1,
        ReplaceInPrimaryHand = wagenT1 holdingtrolleyright,
        ReplaceInSecondHand  = wagenT1 holdingtrolleyleft,
        RunSpeedModifier = 0.80,
        Tags             = Holzwagen;HolzwagenT1,
        Icon             = HolzwagenT1,
    }

    item Holzwagen_T2
    {
        DisplayName      = Holzwagen (T2),
        ItemType         = base:container,   /* B42-Container-Vorlage: oeffenbar wie ein Rucksack */
        Weight           = 11.0,
        Capacity         = 50,
        WeightReduction  = 100,
        AcceptItemFunction = HolzwagenAccept,
        CloseSound       = CloseBag,
        OpenSound        = OpenBag,
        PutInSound       = PutItemInBag,
        WorldStaticModel = wagenT2world,  /* abgestellt: Handgriff am Boden */
        StaticModel      = wagenT2,
        ReplaceInPrimaryHand = wagenT2 holdingtrolleyright,
        ReplaceInSecondHand  = wagenT2 holdingtrolleyleft,
        RunSpeedModifier = 0.90,
        Tags             = Holzwagen;HolzwagenT2,
        Icon             = HolzwagenT2,
    }

    /* Bau-Komponenten */
    item Holzwagen_RadT1
    {
        DisplayName = Wagenrad (Holz/Stein),
        Type = Normal, Weight = 3.0,
        Tags = HolzwagenRad;HolzwagenRadT1,
        Icon = HolzwagenRadT1,
    }
    item Holzwagen_Speichengestell
    {
        DisplayName = Speichenrad-Gestell (roh),
        Type = Normal, Weight = 2.5,
        Tags = HolzwagenRadGestell,
        Icon = HolzwagenGestell,
    }
    item Holzwagen_RadT2
    {
        DisplayName = Speichenrad (beschlagen),
        Type = Normal, Weight = 3.5,
        Tags = HolzwagenRad;HolzwagenRadT2,
        Icon = HolzwagenRadT2,
    }
}
