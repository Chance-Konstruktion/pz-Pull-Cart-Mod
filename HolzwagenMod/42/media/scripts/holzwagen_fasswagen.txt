module Base
{
    /* Eigenes Modell: Wagen MIT Fass/Schlauch/Trichter.
       Mesh kommt aus holzwagen_fass_blender.py -> holzwagen_fass.fbx
       in 42/media/models_X/. Textur wird mitbenutzt (gleicher Atlas). */
    model wagenFass
    {
        mesh = holzwagen_fass, texture = holzwagen_tex, scale = 0.007,
        /* In-Hand: Handgriff hoch Richtung Haende (a = -20). Boden: Handgriff
           kippt auf der Radachse nach unten (attachment world, a = 25). */
        attachment Bip01_Prop1 { offset = -1.4 0.1 0.0, rotate = -20.0 90.0 0.0, }
        /* Boden-Kippung folgt als gebackenes Welt-Modell (wie T2); attachment-
           world-rotate wuerde die Raeder versenken -> entfernt. */
    }

    /* =========================================================
       FASSWAGEN – Holzwagen mit fest verbautem Fass.
       Transportiert Fluessigkeiten; Bett vom Fass belegt, dafuer 4 Taschen-Slots.
       Fluid-Syntax verifiziert am Spielcode (Build 42.18: WaterBottle/GasCan).
       ========================================================= */
    item Holzwagen_Fasswagen
    {
        DisplayName      = Fasswagen,
        Type             = Container,      /* haelt die seitlichen Taschen-Slots */
        Weight           = 16.0,
        Capacity         = 50,
        WeightReduction  = 100,
        AcceptItemFunction = HolzwagenAccept,  /* Bett gesperrt: nur Taschen, kein loser Loot */
        WorldStaticModel = wagenFass,
        StaticModel      = wagenFass,
        ReplaceInPrimaryHand = wagenFass holdingtrolleyright,
        ReplaceInSecondHand  = wagenFass holdingtrolleyleft,
        RunSpeedModifier = 0.75,
        Tags             = Holzwagen;HolzwagenFass,
        Icon             = Crate,

        /* Das fest verbaute Fass als Fluid-Container.
           Capacity in Fluid-Einheiten (Wasserflasche = 1.0 ~ 1 Liter Basis).
           450 = grosser Tank; bei Bedarf runterdrehen. */
        component FluidContainer
        {
            ContainerName = Holzwagen_Fass,
            /* B42 deckelt Fluid-Container offenbar bei ~142 L (Test: Fass wurde
               nur bis 141,6 L voll). 450 war daher unerreichbar -> auf 140
               gesetzt, damit der Tank sauber voll wird und ein voller Wagen
               nicht absurd schwer ist (Grund, warum voll schwer abzustellen war). */
            Capacity = 140.0,
            /* KEIN Fluids-Startinhalt -> Fass startet LEER.
               Die alte Fluids{}-Liste (fluid = Water:1.0 usw.) hat das Fass
               beim Erstellen vorbefuellt (halb voll). Ein leerer FluidContainer
               nimmt die Standard-Fluids beim Befuellen an.
               Wenn wir spaeter nur bestimmte Fluids zulassen wollen, kommt die
               Whitelist hier wieder rein - aber mit Ratio 0, nie als Startinhalt. */
        }
    }

    /* ---- Komponenten (selbst definiert, damit keine ungueltigen Vanilla-IDs den Mod brechen) ---- */
    item Holzwagen_Fass
    {
        DisplayName = Fass (leer),
        Type = Normal, Weight = 6.0,
        Tags = HolzwagenFassItem,
        Icon = Crate,
    }
    item Holzwagen_Schlauch
    {
        DisplayName = Schlauch,
        Type = Normal, Weight = 0.5,
        Tags = HolzwagenSchlauch,
        Icon = Hose,
    }
    item Holzwagen_Trichter
    {
        DisplayName = Trichter,
        Type = Normal, Weight = 0.4,
        Tags = HolzwagenTrichter,
        Icon = Funnel,
    }

    /* =========================================================
       REZEPTE  (B42-Minimalformat, verifiziert; Skill/Tools spaeter)
       ========================================================= */
    /* TEST-Rezept: Fasswagen direkt bauen wie der T1-Wagen (1 Baumstamm +
       Hammer, Woodwork 1), damit er schnell testbar ist. Spaeter durch den
       richtigen Umbau-Weg (T2 + Fass) ersetzen/ergaenzen. */
    craftRecipe Holzwagen_Fasswagen_Test
    {
        timedAction = Making,
        time = 120,
        Tags = AnySurfaceCraft;Carpentry,
        category = Carpentry,
        SkillRequired = Woodwork:1,
        xpAward = Woodwork:10,
        inputs
        {
            item 1 tags[base:hammer] mode:keep flags[MayDegradeVeryLight],
            item 1 [Base.Log] flags[Prop2],
        }
        outputs
        {
            item 1 Base.Holzwagen_Fasswagen,
        }
    }

    craftRecipe Holzwagen_Fass_Bauen
    {
        time =120,
        category = Carpentry,
        Tags = AnySurfaceCraft;Carpentry,
        timedAction = Making,
        SkillRequired = Woodwork:1,
        xpAward = Woodwork:10,
        inputs
        {
            item 1 [Base.SheetMetal],
        }
        outputs
        {
            item 1 Base.Holzwagen_Fass,
        }
    }

    craftRecipe Holzwagen_Schlauch_Bauen
    {
        time =40,
        category = Carpentry,
        Tags = AnySurfaceCraft;Carpentry,
        timedAction = Making,
        SkillRequired = Woodwork:1,
        xpAward = Woodwork:10,
        inputs
        {
            item 2 [Base.DuctTape],
        }
        outputs
        {
            item 1 Base.Holzwagen_Schlauch,
        }
    }

    craftRecipe Holzwagen_Trichter_Bauen
    {
        time =40,
        category = Carpentry,
        Tags = AnySurfaceCraft;Carpentry,
        timedAction = Making,
        SkillRequired = Woodwork:1,
        xpAward = Woodwork:10,
        inputs
        {
            item 1 [Base.SheetMetal],
        }
        outputs
        {
            item 1 Base.Holzwagen_Trichter,
        }
    }

    craftRecipe Holzwagen_Umbau_Fasswagen
    {
        time =150,
        category = Carpentry,
        Tags = AnySurfaceCraft;Carpentry,
        timedAction = Making,
        SkillRequired = Woodwork:1,
        xpAward = Woodwork:10,
        inputs
        {
            item 1 [Base.Holzwagen_T2],
            item 1 [Base.Holzwagen_Fass],
            item 1 tags[base:hammer] mode:keep flags[MayDegradeVeryLight],

        outputs
        {
            item 1 Base.Holzwagen_Fasswagen,
        }
    }

    craftRecipe Holzwagen_Rueckbau_Fasswagen
    {
        time =80,
        category = Carpentry,
        Tags = AnySurfaceCraft;Carpentry,
        timedAction = Making,
        SkillRequired = Woodwork:1,
        xpAward = Woodwork:10,
        inputs
        {
            item 1 [Base.Holzwagen_Fasswagen],
            item 1 tags[base:hammer] mode:keep flags[MayDegradeVeryLight],
        }
        outputs
        {
            item 1 Base.Holzwagen_T2,
            item 1 Base.Holzwagen_Fass,
        }
    }
}
